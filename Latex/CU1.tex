\begin{UseCase}{CU1}{Iniciar Sesión}{
		\noindent Este caso de uso permite que el actor alumno pueda ingresar a la aplicación dandole las opciones de inscribirse en algún evento interpolitécnico deportivo de su interés o consultar los eventos a los que ya se ha registrado previamente. \\
    	Para poder iniciar sesión el alumno deberá oprimir el botón \IUbutton{ Iniciar Sesión } ubicado en la pantalla ... Ingresará su número de boleta, contraseña el cual usa para ingresar al SAES (Sistema de Administración Escolar), y el captcha, como se muestra en la pantalla... Si los datos que ingresa no coinciden se le mostrará un mensaje.
        Una vez que incie sesión se le mostrará la pantalla principal.
	} \label{CU1_Iniciarsesion}
		\UCitem{Versión}{0.1}
		\UCitem{Autor}{Rosales González Carlos Andrés}
		\UCitem{Supervisa}{Mendoza García Bruno Alejandro}
		\UCitem{Actor}{Alumno}
		\UCitem{Propósito}{Tener control de las personas registradas.}
        \UCitem{Precondiciones}{
        \begin{itemize}
            \item Ser alumno del Instituto Politécnico Nacional.
            \item Contar con el número de boleta usada en el SAES.
            \item Contar con la contraseña usada en el SAES.
        \end{itemize}}
        \UCitem{Postcondiciones}{Se muestra la pantalla principal}
		\UCitem{Entradas}{
        \begin{itemize}
        	\item Número de boleta 
        	\item Contraseña
        	\item Captcha
        \end{itemize}}
		\UCitem{Origen}{Pantalla, Teclado}
		\UCitem{Salidas}{
		\begin{itemize}
		    \item Acceso a la página principal del alumno
		\end{itemize}}
		\UCitem{Destino}{Pantalla}
		\UCitem{Errores}{
        	\begin{itemize}
        	    \item Los campos están vacíos.
            	\item Boleta y/o contraseña incorrecta. Mensaje .
            	\item Captcha incorrecto.
            \end{itemize}
       }
		\UCitem{Observaciones}{}
		\end{UseCase}
    \begin{UCtrayectoria}{Principal}
    \UCpaso[\UCactor] Oprime el botón \IUbutton{ Iniciar Sesión } que esta en la pantalla \IUref{p19Iniciogeneral}{Pantalla principal general \ref{Iniciogeneral1}. }
    \UCpaso Muestra la pantalla \IUref{p1Login}{Pantalla de Inicio de Sesión \ref{Login}.}
	\UCpaso[\UCactor] Introduce Boleta, contraseña y captcha. \label{CU1_regresar} 
    \UCpaso[\UCactor] Presiona el botón \IUbutton{ Ingresar }.
    \UCpaso Comprueba que los campos no estén vacíos. \Trayref{A} \Trayref{B}
    \UCpaso Obtiene los valores ingresados
    \UCpaso Válida campos. 
    \UCpaso Muestra la pantalla \IUref{p1Login}{Pantalla de Inicio de Sesión. \ref{Inicioparticipante}.}
    \end{UCtrayectoria}
    
	\begin{UCtrayectoriaA}{A}{Boleta y/o contraseña erróneo}
		\UCpaso muestra mensaje “Error: La boleta y/o contraseña que se ingresó son erróneos”. Mensaje .
   		\UCpaso Continua en el paso \ref{CU1_regresar} del \UCref{CU1}.
	\end{UCtrayectoriaA}
	
	\begin{UCtrayectoriaA}{B}{El captcha no coincide}
		\UCpaso muestra mensaje “El captcha que se ingreso no coincide”. Mensaje .
   		\UCpaso Continua en el paso \ref{CU1_regresar} del \UCref{CU1}.
	\end{UCtrayectoriaA}
